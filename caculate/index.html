理论

$$ \begin{align} &\text{投资回报率}(ROI)=\text{净利润/前期总成本}\times 100\%\ &\text{前期总成本}=\text{总开发成本}+\text{总开发费用}+\text{土地金额}+\text{其他扣除项目}\ &k=\frac{\text{普通宅面积}}{\text{非普通宅面积}+\text{普通宅面积}}=\frac{\sum_{i\in \text{普通宅}}\text{房型面积}{i}\times \text{建房套数}{i}}{\sum_{i\in \text{非普通宅}}\text{房型面积}{i}\times \text{建房套数}{i}+\sum_{i\in \text{普通宅}}\text{房型面积}{i}\times \text{建房套数}{i}}\ &\text{总开发成本}=\sum_{i}\text{开发成本}{i}\times \text{建房套数}{i}\times \text{房型面积}{i}\ &\text{总允许开发成本}=\sum{i\in \text{允许扣除}}\text{开发成本}{i}\times \text{建房套数}{i}\times \text{房型面积}{i}\ &\text{总土地金额}=777179627(given)\ &\text{普通住房收入}=\sum{i\in \text{普通宅}}\text{房型面积}{i}\times \text{建房套数}{i}\times \text{售价}{i}\ &\text{非普通住房收入}=\sum{i\in \text{非普通宅}}\text{房型面积}{i}\times \text{建房套数}{i}\times \text{售价}{i}\ &\text{其他住房收入}=\sum{i\in \text{其他}}\text{房型面积}{i}\times \text{建房套数}{i}\times \text{售价}_{i}\ &\text{折合普通住房收入（含其他）}=\text{普通住房收入}+k\times \text{其他住房收入}\ &\text{折合非普通住房收入（含其他）}=\text{非普通住房收入}+(1-k)\times \text{其他住房收入}\ &\text{总收入}=\text{普通住房收入}+\text{非普通住房收入}+\text{其他住房收入}\ &\text{总开发费用}=0.1\times (\text{总开发成本}+\text{土地金额})\ &\text{总允许开发费用}=0.1\times (\text{总允许开发成本}+\text{土地金额})\ &\text{总其他扣除费用}=2\times \text{开发费用}\ &\text{总转让税金}=0.0565\times \text{总收入}\ &\text{总扣除金额}=\text{总允许开发成本}+\text{总土地金额}+\text{总允许开发费用}+\text{总其他扣除项目}+\text{总转让税金}\ &\text{总增值额}=\text{总收入}-\text{总扣除金额}\ \end{align} $$

$$ \text{土地增值税（普通宅）}= \left{ \begin{array}{ll} 0 & (\frac{\text{增值税}}{\text{扣除项目金额}}\leq 0.2)\ \text{增值税}\times 0.3 & (0.2<\frac{\text{增值税}}{\text{扣除项目金额}}\leq 0.5)\ \text{增值税}\times 0.4-\text{扣除项目金额}\times 0.05 & (0.5<\frac{\text{增值税}}{\text{扣除项目金额}}\leq 1)\ \text{增值税}\times 0.5-\text{扣除项目金额}\times 0.15 & (1<\frac{\text{增值税}}{\text{扣除项目金额}}\leq 2)\ \text{增值税}\times 0.6-\text{扣除项目金额}\times 0.35 & (\frac{\text{增值税}}{\text{扣除项目金额}}>2)\ \end{array} \right. $$ $$ \text{土地增值税（非普通宅）}= \left{ \begin{array}{ll} \text{增值税}\times 0.3 & (\frac{\text{增值税}}{\text{扣除项目金额}}\leq 0.5)\ \text{增值税}\times 0.4-\text{扣除项目金额}\times 0.05 & (0.5<\frac{\text{增值税}}{\text{扣除项目金额}}\leq 1)\ \text{增值税}\times 0.5-\text{扣除项目金额}\times 0.15 & (1<\frac{\text{增值税}}{\text{扣除项目金额}}\leq 2)\ \text{增值税}\times 0.6-\text{扣除项目金额}\times 0.35 & (\frac{\text{增值税}}{\text{扣除项目金额}}>2)\ \end{array} \right. $$
